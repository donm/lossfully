bitrate_kbs(path)

Return the bitrate of the file as an integer in kbs (‘soxi -B’). If soxi does not recognize the file as audio, return nil.

[Source]

    def self.bitrate_kbs path
      #(soxi_command path, '-D').to_f
      b = bitrate(path)
      return b.to_f * (b[-1..-1] == 'k' ? 1 : 1000)
    end

soxi_command(path, options='')

# File lib/lossfully/audiofile.rb, line 16
    def self.soxi_command path, options=''
      check_file path

      # system("soxi -V0 #{path}")
      p = IO.popen("soxi -V0 #{options} #{path}")
      return ((Process.wait2 p.pid)[1] == 0) ? p.gets.chomp : nil
    end
